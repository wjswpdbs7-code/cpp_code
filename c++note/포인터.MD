#include <iostream>
using namespace std;

class Circle{
    int radius;
public:
    Circle() { radius = 1; }
    Circle(int r) { radius = r; }
    double getArea();
};

double Circle::getArea(){
    return 3.14 * radius * radius;
}

int main(){
    Circle dount;
    Circle pizza(30);

    cout <<  dount.getArea() << endl; 

    Circle *p;
    p = &dount;
    cout << p->getArea() << endl;
    cout << (*p).getArea() << endl;

    p = &pizza;
    cout << p->getArea() << endl;
    cout << (*p).getArea() << endl;
}

해당 예제로 클래스에서의 포인터 사용법 설명


this 포인터는 클래스의 멤버 함수에서 객체 자신을 가리키는 포인터입니다. 멤버 함수가 호출될 때, this 포인터는 해당 함수를 호출한 객체의 주소를 가리킵니다. 이를 통해 멤버 함수 내에서 객체의 멤버 변수나 다른 멤버 함수에 접근할 수 있습니다.

예시

#include <iostream> 
using namespace std;
class Circle {
    int radius;
public:
    Circle() { radius = 1; }
    Circle(int r) { radius = r; }
    double getArea() {
        return 3.14 * radius * radius;
    }
    void setRadius(int r) {
        this->radius = r; // this 포인터를 사용하여 멤버 변수 radius에 접근
    }
};




C++에서 직접 포인터 대신 참조(reference)를 쓰면 문법이 더 깔끔합니다: void swap(int &a, int &b) { int temp=a; a=b; b=temp; }
이미 표준에 있는 std::swap을 쓰면 안전하고 간편합니다.
포인터를 사용할 때는 null(0) 주소 체크가 필요할 수 있습니다(여기 예제에서는 &m, &n으로 null 가능성이 없음).




int main(){
    int b = 10;
    int &a;

    a = b; 

    cout << a << endl;
    
    cout << b << endl;
    return 0;
}

이니셜라이저는 참조자 선언과 동시에 초기화해야 합니다. 참조자는 선언과 동시에 초기화되어야 하며, 이후에는 다른 변수나 값으로 변경할 수 없습니다. 따라서 int &a;와 같이 참조자를 선언만 하고 초기화하지 않으면 컴파일 오류가 발생합니다. 참조자는 반드시 선언과 동시에 초기화되어야 합니다.
