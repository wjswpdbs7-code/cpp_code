# 객체, 클래스


객체 클래스의 예시 코드
```cpp
#include <iostream>

using namespace std;

class Car {
public:
    string brand;// 멤버 변수
    string model;// 멤버 변수
    int year;// 멤버 변수

    void start() {
        cout << "The car has started." << endl;
    }
};

int main() {
    Car myCar; // Car 클래스의 객체 생성
    myCar.brand = "Toyota"; // 객체의 멤버 변수에 값 할당
    myCar.model = "Corolla";
    myCar.year = 2020;

    cout << "Brand: " << myCar.brand << endl; // 객체의 멤버 변수 출력
    cout << "Model: " << myCar.model << endl;
    cout << "Year: " << myCar.year << endl;

    myCar.start(); // 객체의 멤버 함수 호출

    return 0;
}
```

생성자 예시

```cpp
#include <iostream>
using namespace std;

class Car {
public:
    string brand;
    string model;
    int year;
    // 생성자 정의
    Car(string b, string m, int y) {
        brand = b;
        model = m;
        year = y;
    }
    void start() {
        cout << "The car has started." << endl;
    }
};
int main() {
    Car myCar("Toyota", "Corolla", 2020); // 생성자를 사용하여 객체 생성

    cout << "Brand: " << myCar.brand << endl; // 객체의 멤버 변수 출력
    cout << "Model: " << myCar.model << endl;
    cout << "Year: " << myCar.year << endl;

    myCar.start(); // 객체의 멤버 함수 호출

    return 0;
}
```
